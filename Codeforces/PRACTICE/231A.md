# A. Teams

[Problem Link](https://codeforces.com/problemset/problem/231/A)

``` c++
#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int t, count = 0;
	cin >> t;

	for (int i = 0; i < t; i++) {
		int arr[3], sum = 0;
		for (int i = 0; i < 3; i++) {
			cin >> arr[i];
			sum += arr[i];
		}
		if (sum >= 2) {
			count++;
		} else {
			continue;
		}
	}

	cout << count;

	return 0;
}
```
